on_actions = {
	on_startup = {
		effect = {
			UNC = {
				country_lock_all_division_template = yes
				set_state_owner = 630
			}
			REB = {
				remove_state_claim = 630
			}
			
		}
	}
	on_daily = {
		effect = {
			for_each_scope_loop = {
				array = global.countries
				if = {
					limit = {
						OR = {
							has_wargoal_against = UNC
						}
					}
					remove_wargoal = {type = all target = UNC}
				}
			}
		}
	}
}
