	#corporate influence increase stuff
	on_weekly = { 
		effect = {
			if = {
				limit = {
					original_tag = SYL
					SYL_CORPORATE_INFLUENCE_ENABLED = yes
				}
				SYL = {
					add_to_variable = {
						var = education_influence
						value = education_influence_growth
					}
				
					add_to_variable = {
						var = construction_influence
						value = construction_influence_growth
					}
				
					add_to_variable = {
						var = production_influence
						value = production_influence_growth

						var = civilian_influence

						remove_ideas = {
							SYL_75EducationInfluence

						limit = { 
							
							check_variable = {
								SYL.education_influence = 100
							}
						}
						add_ideas = {
							SYL_100EducationInfluence
							
						}
						remove_ideas = {
							SYL_90EducationInfluence
						}
					}
				}
				#construction related
				SYL = {
					if = {
						limit = { 
							check_variable = {
								SYL.construction_influence > 4
							}
							check_variable = {
								SYL.construction_influence < 25

							SYL_25ConstructionInfluence

							check_variable = {
								SYL.construction_influence > 24
							}
							check_variable = {
								SYL.construction_influence < 50
							}
						}
						add_ideas = {
							SYL_50ConstructionInfluence
						}
						remove_ideas = {
							SYL_25ConstructionInfluence
							SYL_75ConstructionInfluence
						}
					}
					if = {
						limit = { 
							check_variable = {
								SYL.construction_influence > 49
							}
							check_variable = {
								SYL.construction_influence < 75
							}
						}
						add_ideas = {
							SYL_75ConstructionInfluence
						}
						remove_ideas = {
							SYL_50ConstructionInfluence
							SYL_90ConstructionInfluence
						}
					}

						limit = { 
							check_variable = {
								SYL.construction_influence > 74

							check_variable = {
								SYL.construction_influence < 100

						}
					}
					if = {
						limit = { 
							
							check_variable = {
								SYL.construction_influence = 100
							}
						}
						add_ideas = {
							SYL_100ConstructionInfluence
							
						}
						remove_ideas = {
							SYL_90ConstructionInfluence
						}
					}
				}

				

			}
			
		}
	}

	on_monthly = {
		effect = {
			if = {
				limit = { original_tag = SYL
				SYL_CORPORATE_INFLUENCE_ENABLED = yes
			}

			SYL = {
				add_to_variable = {
					var = education_influence_growth
					value = 0.5
				}
			
				add_to_variable = {
					var = construction_influence_growth
